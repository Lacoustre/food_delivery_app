rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to read/write their own driver document
    match /drivers/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow phone number uniqueness checking and linking
    match /phone_index/{phoneNumber} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && 
        request.auth.uid == resource.data.uid;
      allow write: if request.auth != null && 
        request.auth.uid == resource.data.uid;
    }
    
    // Allow email-phone linking
    match /email_phone_links/{email} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && 
        request.auth.uid == resource.data.uid;
      allow write: if request.auth != null && 
        request.auth.uid == resource.data.uid;
    }
    
    // Orders collection - drivers can read/write their own orders
    match /orders/{orderId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.driverId || 
         request.auth.uid == resource.data.customerId);
      allow create: if request.auth != null;
    }
    
    // Default deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
